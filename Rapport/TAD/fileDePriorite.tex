  \tadNom{fileDePriorité}
  \tadDependances{Booleen, Elément}
  \begin{tadOperations}{fileDePriorité}

    \tadOperation{estVide}{\tadUnParam{fileDePriorité}}{\tadUnParam{Booleen}}
    \tadOperation{défiler}{\tadUnParam{fileDePriorité}}{\tadUnParam{fileDePriorité}}

    \end{tadPrecondition}

  \end{tadPreconditions}
