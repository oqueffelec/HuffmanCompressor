\begin{algorithme}
\type{FileDePriorite}{FDP\_Noeud}
  \begin{enregistrement}{FDP\_Noeud}
    \item arbre :  ArbreDeHuffman
    \item fileSuivante : FileDePriorite
  \end{enregistrement}\\
  \fonction{fileDePriorite}{}{FileDePriorite}
  {}
  {
    \retourner{NIL}
  }
\end{algorithme}

\begin{algorithme}
  \procedure{estVide}{
  \paramEntree{fdp : FileDePriorite}
  \paramSortie{\booleen}
  }
  {
  }
  {
  \retourner{fdp=NIL}
  }
\end{algorithme}

\begin{algorithme}
  \procedure{enfilerADH}{
  \paramEntree{a: ArbreDeHuffman}
  \paramEntreeSortie{fdp : FileDePriorite}
  }
  {
  temp : FileDePriorite
  }
  {
  \affecter{temp}{fileDePriorite}
  \sialorssinon{estVide(fdp) ou obtenirPonderation(obtenirADH(fdp))>= obtenirPonderation(a)}{
  \instruction{ajouter(fdp,a)}
  }
  {
  \affecter{temp}{obtenirFileSuivante(fdp)}
  \instruction{enfilerADH(temp,a)}
  \instruction{fixerFileSuivante(fdp,temps)}
  }
  }
\end{algorithme}

\begin{algorithme}
  \fonctionAvecPreconditions{obtenirADH}{fdp: FileDePriorite}{ArbreDeHuffman}
  {
  nonestvide(fdp)
  }
  {}
  {
 \retourner{fdp.arbre}
  }
\end{algorithme}

\begin{algorithme}
  \fonction{longueur}{fdp: FileDePriorite}{\naturel}
  {
  res : \naturel \\
  temp : FileDePriorite
  }
  {
  \affecter{res}{0}
   \affecter{temp}{fdp}
   \tantque{non(estVide(temp))}{
   	\affecter{temp}{obtenirFileSuivante(temp)}
	\affecter{res}{res+1}
   }
 \retourner{fdp.arbre}
  }
\end{algorithme}

\begin{algorithme}
  \procedureAvecPreconditions{defilerADH}{
  \paramEntreeSortie{fdp: FileDePriorite}
  }
  {non(estVide(fdp))}
  {
  temp : FileDePriorite
  }
  {
  \affecter{temp}{fdp}
  \affecter{fdp}{fdp.listeSuivante}
  \instruction{liberer(temp)}
  }
\end{algorithme}
