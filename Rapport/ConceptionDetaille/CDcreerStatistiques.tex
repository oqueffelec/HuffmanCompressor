\begin{algorithme}
  \fonction{creerStatistiques}{fichier : FichierBinaire}{Statistiques}
  {
  stat : Statistiques \\
  octet : Octet
	}
  {
  \affecter{octet}{octet()}
  \affecter{stat}{statistiques()}
  \tantque{lireOctet(fichier,octet)=1 ET finfichier(fichier)=0}{
  \instruction{ajouter(stat,octet)}
  }
  \retourner{stat}
  }
\end{algorithme}
