\begin{algorithme}
  \begin{enregistrement}{Statistiques}
    \item valeur[STATSIZE] : naturel
  \end{enregistrement}\\
  \fonction{statistiques}{}{Statistiques}
  {
	stat : Statistiques\\
	i : \naturel\\

	}
  {

		\pour{i}{0}{STATSIZE}{}
			{
				\affecter {stat.valeur[i]}{0}
			}
	\retourner{stat}
  }
\end{algorithme}

\begin{algorithme}
  \procedure{ajouter}{
  \paramEntreeSortie{stat : Statistiques}
  \paramEntree{octet : Octet}
  }
  {}
  {
  \affecter{stat.valeur[octetEnDecimal(octet)]}{obtenirPonderation(stat,octet)+1}
  }
\end{algorithme}


\begin{algorithme}
  \fonction{estPresentOctet}{stats : Statistiques, o : Octet}{\booleen}
  {
	estPresent : \booleen
	}
  {
	\affecter{estPresent}{FAUX}

  \sialors{obtenirPonderation(stat,o) $\neq$ 0}
    {
    \affecter{estPresent}{VRAI}
    }

\retourner{estPresent}
  }
\end{algorithme}

\begin{algorithme}
  \fonction{obtenirPonderation}{stats : Statistiques, o : Octet}{\naturel}
  {}
  {
\retourner{stat.valeur[octetEnDecimal(o)]}
  }
\end{algorithme}
